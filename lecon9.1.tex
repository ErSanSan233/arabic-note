\chapter{君の名は？}

\lecon{9.1}

\section{单词和课文}

\begin{itemize}
    \item \ac{مَا}{谁}
    \item \ac{اِسْمٌ}{名字}
    \item \ac{اِسْمُكَ / اِسْمُكِ}{你的名字}
    \item \ac{اِسْمِى}{我的名字}
    \item \ac{اِسْمُهُ / اِسْمُهَا}{他的名字}
    \item \ac{بِ}{prep. 借助}
    \item \ac{أَهْلََا بِكَ!}{你好}
    \item \ac{حَسَنََا}{好的}
    \item \ac{إِلَى}{prep. 向、到}
    \item \ac{اَلْ}{冠词（无\arm{ء
    }）}
    \item \ac{لِقَاءٌ}{n. 见面}
    \item \ac{إِلَى اللِّقَاءِ}{再见}
\end{itemize}

\begin{Arabic}
    - أَهْلاََ وَسَهْلاََ! 

    - أَهْلاََ وَسَهْلاََ!

    - مَا اسْمُكَ؟

    - اِسْمِى أَهْمَدُ. وَمَا اسْمُكَ؟

    - اِسْمِى مَنْصُورٌ.

    - أَهْلاََ وَسَهْلاََ!

    - أَهْلاََ بِكَ!

    - أَنَا مَنْصُورٌ. وَمَا اسْمُكِ؟

    - اِسْمِى يُمْنَى.

    - حَسَنََا.إِلَى للِّقَاءِ!

    - إِلَى للِّقَاءِ! 
\end{Arabic}

\begin{note}
    \arm{مَا اسْمُكَ؟}读音为\arm{مَسْمُكَ}。

    \arm{إِلَى} （介词）后面 \arm{لِقَاءٌ} 变为属格 \arm{لِقَاءِ} 。
\end{note}
