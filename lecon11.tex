\chapter{你是哪国人}

\section{\lecon{11.1}询问国籍}

\begin{itemize}
    \item \ac{جِنْسِيَّةٌ}{国籍}
\end{itemize}

末尾的 \arm{ـة}标记了阴性名词。

\begin{Arabic}
    \begin{itemize}
        \item مَا جِنْسِيَّتُكَ/جِنْسِيَّتُكِ؟
        \item مَا جِنْسِيَّتُهُ/جِنْسِيَّتُهَا؟
    \end{itemize}
\end{Arabic}
